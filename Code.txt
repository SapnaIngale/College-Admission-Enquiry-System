/* LOGIN */
function login() {
  let u = loginUser.value;
  let p = loginPass.value;
  if(u === "admin" && p === "admin123") {
    loginPage.style.display = "none";
    homePage.style.display = "block";
  } else {
    alert("Invalid Login! Use admin / admin123");
  }
}

/* LOGOUT */
function logout() {
  homePage.style.display = "none";
  loginPage.style.display = "flex";
}

/* COURSE FEES */
course.addEventListener("change", function() {
  let feesMap = {
    "BCA":"?60,000",
    "B.Sc IT":"?55,000",
    "MCA":"?80,000",
    "B.Tech":"?1,20,000",
    "Diploma CS":"?40,000"
  };
  fees.value = feesMap[this.value] || "";
});

/* SUBMIT FORM */
function submitForm(e) {
  e.preventDefault();

  let enquiry = {
    name: name.value,
    phone: phone.value,
    course: course.value,
    email: email.value
  };

  let data = JSON.parse(localStorage.getItem("enquiries")) || [];
  data.push(enquiry);
  localStorage.setItem("enquiries", JSON.stringify(data));

  alert("Enquiry Submitted Successfully!");
  e.target.reset();
}

/* VIEW ENQUIRIES */
function viewEnquiries() {
  let data = JSON.parse(localStorage.getItem("enquiries")) || [];
  if(data.length === 0) {
    alert("No enquiries found.");
    return;
  }
  let msg = "ENQUIRIES LIST:\n\n";
  data.forEach((e,i) => {
    msg += (i+1)+". "+e.name+" - "+e.course+"\n";
  });
  alert(msg);
}
